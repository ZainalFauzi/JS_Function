2.Buat sebuah function helper untuk convert temperatur Celcius ke Fahrenherit dan sebaliknya. Ikuti poin-poin dibawah ini: Buat sebuah function helper lalu gunakan pada function converternya. Seperti contoh dibawah ini. Jadi kita tinggal membuat function helper dan function utama untuk convert fahrenheit ke Celcius 
            const kaliSembilanPerLima = (number1) => {
                return number1 * (9 / 5);
              };
              
              const getFahrenheit = (celcius) => {
                console.log("menjadi ", kaliSembilanPerLima(celcius) + 32, " fahrenheit");
              };
              
              getFahrenheit(15);
              
              const kurangTigaPuluhDua = (number2) => {
                return number2 - 32;
              };
              
              const getCelcius = (fahrenheit) => {
                console.log("menjadi ", (kurangTigaPuluhDua(fahrenheit) * 5) / 9, " celcius");
              };
              
              getCelcius(32);
